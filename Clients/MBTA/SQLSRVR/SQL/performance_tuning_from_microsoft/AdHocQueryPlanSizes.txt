Select 'Total Entries'
select      COUNT(*) As NumberOfPlans, 
            SUM(size_in_bytes) / 1024.0/1024.0 AS SizeInMB
from  sys.dm_exec_cached_plans

Select 'Total Entries Broken Out by Type'
select COUNT(*) As NumberOfPlans, 
            SUM(size_in_bytes) / 1024.0/1024.0 AS SizeInMB,
            objtype
from  sys.dm_exec_cached_plans
group by objtype

Select 'Those only used once'
select      COUNT(*) As NumberOfPlans, 
            SUM(size_in_bytes) / 1024.0/1024.0 AS SizeInMB
from  sys.dm_exec_cached_plans
where usecounts = 1

Select 'Those only used once Broken Out by Type'
select      COUNT(*) As NumberOfPlans, 
            SUM(size_in_bytes) / 1024.0/1024.0 AS SizeInMB,
            objtype
from  sys.dm_exec_cached_plans
where usecounts = 1
group by objtype
